        function checkStatus(fieldKey, value, gender) { 
            let r = REF_DATA[fieldKey]; 
            if (!r || !value || value === "" || value === "-") return ""; 
            
            if (fieldKey === 'u_warna') return ""; // Menghilangkan keterangan khusus warna urine
            
            let isMale = gender && gender.toLowerCase().includes("laki");
            let isFemale = gender && gender.toLowerCase().includes("perempuan");

            if (r.male && isMale) r = { ...r, ...r.male };
            if (r.female && isFemale) r = { ...r, ...r.female };
            
            const num = parseFloat(value.toString().replace(',', '.')); 
            if (isNaN(num)) return (r.normal && String(value).toLowerCase() !== String(r.normal).toLowerCase()) ? "TIDAK NORMAL" : "NORMAL"; 
            if (r.range && (num < r.range[0])) return "RENDAH"; 
            if (r.range && (num > r.range[1])) return "TINGGI"; 
            if (r.max && num > r.max) return "TINGGI"; 
            if (r.min && num < r.min) return "RENDAH"; 
            return "NORMAL"; 
        }
        
        function getRefString(fieldKey, gender) { 
            let r = REF_DATA[fieldKey]; 
            if (!r) return "-"; 
            if (r.male && r.female) {
                let m = r.male.range ? r.male.range[0] + " - " + r.male.range[1] : (r.male.max ? "< " + r.male.max : "");
                let f = r.female.range ? r.female.range[0] + " - " + r.female.range[1] : (r.female.max ? "< " + r.female.max : "");
                return "L: " + m + "<br>P: " + f + (r.unit ? " " + r.unit : ""); 
            }
            let s = ""; 
            if (r.range) s = r.range[0] + " - " + r.range[1]; 
            else if (r.max) s = "< " + r.max; 
            else if (r.min) s = "> " + r.min; 
            else if (r.normal) s = r.normal; 
            return s + (r.unit ? " " + r.unit : ""); 
        }
